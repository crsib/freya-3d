00162         void generateFrustum()
00163         {
00164                 qcheck(mDir);
00165                 float halfAngle = 0.5f * mFOW * math::pi / 180.0;
00166                 float fovValueY = 1 / (float)(tan(halfAngle));
00167                 float fovValueX = fovValueY / mAspect;
00168                 math::vector3d v(mDir * mFar);
00169                 math::vector3d hr(mRight * (mFar / fovValueX));
00170                 math::vector3d hu(mUp * (mFar / fovValueY));
00171                 math::vector3d vv[4];
00172                 vv[0] = v + hr + hu;
00173                 vv[1] = v - hr + hu;
00174                 vv[2] = v - hr - hu;
00175                 vv[3] = v + hr - hu;
00176                 math::plane near(mDir, mNear * mDir);
00177                 math::plane far(mDir, mFar * mDir);
00178                 delete mFrust;
00179 
00180                 mFrust = new camera::Frustum(mPos, near, far);
00181                 mFrust->setVerticies(&vv[0]);
00182                 mFrust->reset();
00183                 mFrust->updateIndexBuffer();
00184                 mFrust->updateVertexBuffer();
